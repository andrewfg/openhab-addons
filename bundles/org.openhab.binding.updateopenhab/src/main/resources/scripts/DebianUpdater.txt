#!/bin/sh

# OpenHAB Auto Update Script for Linux / Debian

# execute command: [EXECUTE_COMMAND]

# stop OpenHAB
echo [PASSWORD] | sudo -S -k systemctl stop openhab.service

# set repository
echo [PASSWORD] | sudo -S -k sh -c 'echo "deb https://openhab.jfrog.io/artifactory/openhab-linuxpkg [TARGET_TYPE] main" > /etc/apt/sources.list.d/openhab.list'

# update OpenHAB
echo [PASSWORD] | sudo -S -k apt-get update
echo [PASSWORD] | sudo -S -k apt-get install openhab

# start OpenHAB
echo [PASSWORD] | sudo -S -k systemctl start openhab.service
